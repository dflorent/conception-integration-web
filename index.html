Tableau simple


<!-- tableau simple -->

<table border="1">
    <tr>
        <td>Nom</td>
        <td>Prénom</td>
    </tr>
    <tr>
        <td>Jean</td>
        <td>Dupond</td>
    </tr>
</table>

Tableau avec en-tête et pied


<!-- tableau avec en-tête et pied -->

<table border="1">
    <thead>
         <tr>
              <th>Nom</td>
              <th>Prénom</td>
         </tr>
    </head>
    <tfoot>
         <tr>
              <td>Nom</td>
              <td>Prénom</td>
         </tr>
    </tfoot> 
    <tbody>
         <tr>
              <td>Jean</td>
              <td>Dupond</td>
         </tr>
    </tbody>
</table>

Fusionner des cellules


<!-- tableaux avec cellules fusionnées -->

<table border="1">
     <tr>
          <td>Nom</td>
          <td>Prénom</td>
     </tr>
     <tr>
          <td colspan="2">Jean Dupont</td>
     </tr>
</table>

<table border="1">
     <tr>
          <td>Nom</td>
          <td rowspan="2">Prénom</td>
     </tr>
     <tr>
          <td>Jean</td>
     </tr>
</table>
